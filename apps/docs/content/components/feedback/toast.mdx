## Placement

Use the `placement` prop on the `Toast.Provider` to set the placement of the Toast. Values can include `top-start`, `top`, `top-end`, `bottom-start`, `bottom`, `bottom-end`.

{['top-start', 'top', 'top-end', 'bottom-start', 'bottom', 'bottom-end'].map((placement, index) => (

  <div key={placement}>
    { index > 0 ? <hr className='x:my-8 nextra-border' /> : <br /> }
    <Usage
      key={placement}
      components={{
        usage: <ToastPlacementExample placement={placement} />,
        wrapper: Wrapper,
      }}
      serializedCode={`
        const ToastExample = () => {
          const [open, setOpen] = React.useState(false);
          return (
            <Toast.Provider placement="${placement}">
              <Button onClick={() => setOpen(true)}>Trigger (${placement})</Button>
              <Toast defaultOpen={false} open={open} onOpenChange={setOpen}>
                <Toast.Indicator />
                <Toast.Content>
                  <Toast.Title>
                    The quick brown fox
                  </Toast.Title>
                  <Toast.Description>
                    Alice was beginning to get very tired of sitting by her sister on the bank, and of having nothing to do.
                  </Toast.Description>
                </Toast.Content>
                <Toast.Close aria-label="Close" />
              </Toast>
              <Toast.Viewport />
            </Toast.Provider>
          )
        };
      `}
    />
  </div>
))}
